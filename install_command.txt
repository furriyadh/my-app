# ==================== أوامر التثبيت السريع ====================

# 1️⃣ تثبيت جميع المكتبات من ملف requirements.txt
pip3 install -r requirements.txt

# 2️⃣ أو تشغيل سكريبت التثبيت التفاعلي
./install_dependencies.sh

# 3️⃣ أو تثبيت المكتبات الأساسية فقط (سريع)
pip3 install fastapi uvicorn pydantic python-multipart google-generativeai google-ads beautifulsoup4 pandas numpy python-dotenv aiohttp requests sqlalchemy redis loguru

# 4️⃣ تثبيت مكتبات الذكاء الاصطناعي فقط
pip3 install google-generativeai scrapegraphai nltk spacy textblob langdetect openai anthropic langchain

# 5️⃣ تثبيت مكتبات Google Ads فقط
pip3 install google-ads google-auth google-auth-oauthlib google-auth-httplib2 google-api-python-client

# 6️⃣ تثبيت مكتبات الويب سكرابينغ فقط
pip3 install scrapegraphai playwright beautifulsoup4 lxml html5lib selenium scrapy requests-html

# ==================== أوامر إضافية ====================

# تحميل نماذج spaCy
python3 -m spacy download en_core_web_sm
python3 -m spacy download ar_core_news_sm

# تحميل بيانات NLTK
python3 -c "import nltk; nltk.download('punkt'); nltk.download('stopwords'); nltk.download('vader_lexicon')"

# تثبيت متصفحات Playwright
playwright install

# ==================== التحقق من التثبيت ====================

# فحص المكتبات المثبتة
pip3 list | grep -E "(fastapi|google|scrapegraph|nltk|spacy)"

# اختبار الاستيراد
python3 -c "import fastapi, google.generativeai, scrapegraphai, nltk, spacy; print('✅ جميع المكتبات مثبتة بنجاح!')"
